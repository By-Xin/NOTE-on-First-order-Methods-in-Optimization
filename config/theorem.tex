      breakable,

      boxrule=0pt,

      after skip=1em,
      title={笔记 \thenote\if\relax\detokenize{#1}\relax\else：#1\fi},
      attach title to upper={\par\vspace{0.35em}},
      coltitle=black,
      fonttitle=\bfseries
    ]%
  }
  {\end{tcolorbox}}
